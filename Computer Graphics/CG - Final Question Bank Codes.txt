#include<windows.h>
#include <GL/glut.h>
#include <math.h>

void show_1() {
	glClearColor(1,1,1,0);
	glClear(GL_COLOR_BUFFER_BIT);
	glColor3f(1,0,0);
	glLineWidth(5);
	glBegin(GL_LINE_STRIP);
	int x=250 , y=250 , r=100;
     float pi = 22.0/7.0;
     for(float i=0 ; i <= 2*pi ; i+=0.001){
     	glVertex2f(x+sin(i)*r, y+cos(i)*(r*2));
	}
	glEnd();
	glFlush();
}
